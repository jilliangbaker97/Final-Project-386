 44def get_options_chain(ticker: str) -> pd.DataFrame:
 45    """Fetch the full options chain for every available expiration date.
 46
 47    Combines calls and puts across all expirations into a single tidy
 48    DataFrame, adding ``expiration``, ``option_type``, and ``ticker`` columns
 49    so downstream code can filter or group without extra joins.
 50
 51    Parameters
 52    ----------
 53    ticker : str
 54        Upper- or lower-case ticker symbol (e.g. ``"AAPL"``).
 55
 56    Returns
 57    -------
 58    pd.DataFrame
 59        Columns mirror yfinance option chain fields plus:
 60        ``expiration`` (str, YYYY-MM-DD), ``option_type`` ("call"|"put"),
 61        ``ticker`` (str).
 62
 63    Raises
 64    ------
 65    ValueError
 66        If the ticker is invalid, has no listed options, or the network
 67        request fails.
 68
 69    Examples
 70    --------
 71    >>> df = get_options_chain("AAPL")
 72    >>> df[df.option_type == "call"].head()
 73    """
 74    ticker = _validate_ticker(ticker)
 75
 76    try:
 77        t = yf.Ticker(ticker)
 78        expirations = t.options
 79    except Exception as e:
 80        raise ValueError(f"Could not fetch options for '{ticker}': {e}") from e
 81
 82    if not expirations:
 83        raise ValueError(f"No options listed for '{ticker}'.")
 84
 85    frames = []
 86    for exp in expirations:
 87        try:
 88            chain = t.option_chain(exp)
 89        except Exception as e:
 90            log.warning("Skipping expiration %s for %s: %s", exp, ticker, e)
 91            continue
 92
 93        for side, df in (("call", chain.calls), ("put", chain.puts)):
 94            if df.empty:
 95                continue
 96            df = df.copy()
 97            df["expiration"] = exp
 98            df["option_type"] = side
 99            df["ticker"] = ticker
100            frames.append(df)
101
102    if not frames:
103        raise ValueError(f"Options chain for '{ticker}' returned no data.")
104
105    result = pd.concat(frames, ignore_index=True)
106    log.info(
107        "get_options_chain('%s'): %d rows across %d expirations.",
108        ticker, len(result), len(expirations),
109    )
110    return result

113def get_price_history(ticker: str, period: str = "1y") -> pd.DataFrame:
114    """Download Open, High, Low, Close, Volume price history for a single ticker.
115
116    Parameters
117    ----------
118    ticker : str
119        Ticker symbol (e.g. ``"SPY"``).
120    period : str, optional
121        Any period string accepted by yfinance: ``"1d"``, ``"5d"``,
122        ``"1mo"``, ``"3mo"``, ``"6mo"``, ``"1y"``, ``"2y"``, ``"5y"``,
123        ``"10y"``, ``"ytd"``, ``"max"``.  Defaults to ``"1y"``.
124
125    Returns
126    -------
127    pd.DataFrame
128        DatetimeIndex with columns: Open, High, Low, Close, Volume,
129        Dividends, Stock Splits, plus a ``ticker`` column for convenience
130        when concatenating multiple symbols.
131
132    Raises
133    ------
134    ValueError
135        If the ticker or period is invalid, or the response is empty.
136
137    Examples
138    --------
139    >>> df = get_price_history("SPY", period="6mo")
140    >>> df[["Open", "Close", "Volume"]].tail()
141    """
142    ticker = _validate_ticker(ticker)
143
144    valid_periods = {
145        "1d", "5d", "1mo", "3mo", "6mo",
146        "1y", "2y", "5y", "10y", "ytd", "max",
147    }
148    if period not in valid_periods:
149        raise ValueError(
150            f"Invalid period '{period}'. Choose from: {sorted(valid_periods)}"
151        )
152
153    try:
154        df = yf.download(ticker, period=period, auto_adjust=True, progress=False)
155    except Exception as e:
156        raise ValueError(f"Download failed for '{ticker}': {e}") from e
157
158    if df.empty:
159        raise ValueError(
160            f"No price data returned for '{ticker}' with period='{period}'."
161        )
162
163    # Flatten MultiIndex columns produced when yfinance returns a single ticker
164    if isinstance(df.columns, pd.MultiIndex):
165        df.columns = df.columns.get_level_values(0)
166
167    df = df.copy()
168    df["ticker"] = ticker
169    log.info(
170        "get_price_history('%s', period='%s'): %d rows, %s to %s.",
171        ticker, period, len(df),
172        df.index.min().date(), df.index.max().date(),

177def get_etf_holdings(ticker: str) -> pd.DataFrame:
178    """Fetch the top holdings and portfolio weights for an ETF via yfinance.
179
180    Uses the yfinance ``Ticker.get_funds_data()`` API to retrieve ETF
181    holdings data.  Falls back to scraping Yahoo Finance if the API
182    method is unavailable.
183
184    Parameters
185    ----------
186    ticker : str
187        ETF ticker symbol (e.g. ``"SPY"``, ``"QQQ"``).
188
189    Returns
190    -------
191    pd.DataFrame
192        Columns: ``ticker`` (the ETF), ``holding_symbol``, ``holding_name``,
193        ``weight_pct`` (float, 0-100).
194
195    Raises
196    ------
197    ValueError
198        If the ticker is invalid or no holdings data can be retrieved.
199
200    Examples
201    --------
202    >>> df = get_etf_holdings("SPY")
203    >>> df.sort_values("weight_pct", ascending=False).head(10)
204    """
205    ticker = _validate_ticker(ticker)
206
207    try:
208        t = yf.Ticker(ticker)
209        funds_data = t.get_funds_data()
210        holdings_df = funds_data.top_holdings
211    except Exception as e:
212        raise ValueError(
213            f"Could not fetch holdings for '{ticker}': {e}"
214        ) from e
215
216    if holdings_df is None or holdings_df.empty:
217        raise ValueError(f"No holdings data returned for '{ticker}'.")
218
219    rows = []
220    for symbol, row in holdings_df.iterrows():
221        name = row.get("Name", row.get("holdingName", ""))
222        weight = row.get("Holding Percent", row.get("holdingPercent", float("nan")))
223        # yfinance returns weight as a fraction (0-1); convert to percentage
224        if pd.notna(weight) and weight <= 1.0:
225            weight = weight * 100
226        rows.append({
227            "ticker": ticker,
228            "holding_symbol": str(symbol),
229            "holding_name": str(name),
230            "weight_pct": round(float(weight), 4) if pd.notna(weight) else float("nan"),
231        })
232
233    if not rows:
234        raise ValueError(f"No holding rows parsed for ETF '{ticker}'.")
235
236    result = pd.DataFrame(rows)
237    log.info(
238        "get_etf_holdings('%s'): %d holdings fetched.", ticker, len(result)
239    )
240    return result

41def get_etf_headlines(ticker: str, max_articles: int = 10) -> list[str]:
42    """Fetch up to *max_articles* recent news headlines for *ticker* via yfinance.
43
44    Parameters
45    ----------
46    ticker : str
47        ETF or stock ticker symbol (e.g. ``"QQQ"``).
48    max_articles : int, optional
49        Maximum number of headlines to return.  Defaults to ``10``.
50
51    Returns
52    -------
53    list[str]
54        Headline strings, potentially empty if no news is available.
55    """
56    t = yf.Ticker(ticker)
57    articles = t.news or []
58    headlines: list[str] = []
59    for article in articles[:max_articles]:
60        # yfinance may nest title under "content" or at top level
61        content = article.get("content") or {}
62        title = (
63            content.get("title")
64            or article.get("title")
65            or article.get("headline", "")
66        )
67        if title:
68            headlines.append(title)
69    return headlines

 72def get_sentiment_score(
 73    ticker: str,
 74    headlines: list[str],
 75    *,
 76    model: str = _DEFAULT_MODEL,
 77    client: anthropic.Anthropic | None = None,
 78) -> int:
 79    """Score recent ETF news headlines on a 1-5 sentiment scale using Claude.
 80
 81    Uses Anthropic **tool use** with ``tool_choice`` to guarantee a structured
 82    integer response.
 83
 84    Parameters
 85    ----------
 86    ticker : str
 87        ETF ticker for context in the prompt.
 88    headlines : list[str]
 89        News headlines to evaluate.
 90    model : str, optional
 91        Anthropic model identifier.  Defaults to Claude Sonnet.
 92    client : anthropic.Anthropic, optional
 93        Pre-configured client.  If ``None``, a new client is created (reads
 94        ``ANTHROPIC_API_KEY`` from the environment).
 95
 96    Returns
 97    -------
 98    int
 99        Sentiment score from 1 (very negative) to 5 (very positive).
100    """
101    if not headlines:
102        return 3  # neutral when no data
103
104    if client is None:
105        client = anthropic.Anthropic()
106
107    prompt = (
108        f"You are a financial analyst. Given these recent news headlines about "
109        f"the {ticker} ETF, rate the overall market sentiment on a scale of "
110        f"1 (very negative) to 5 (very positive). Use the record_sentiment "
111        f"tool to return your score.\n\n"
112        + "\n".join(f"- {h}" for h in headlines)
113    )
114
115    response = client.messages.create(
116        model=model,
117        max_tokens=64,
118        tools=[_SENTIMENT_TOOL],
119        tool_choice={"type": "tool", "name": "record_sentiment"},
120        messages=[{"role": "user", "content": prompt}],
121    )
122
123    for block in response.content:
124        if block.type == "tool_use":
125            return int(block.input["sentiment_score"])
126
127    return 3  # fallback: neutral

24def clean_prices(df: pd.DataFrame) -> pd.DataFrame:
25    """Standardize a raw price-history DataFrame.
26
27    Parameters
28    ----------
29    df : pd.DataFrame
30        Raw output from :func:`get_price_history` (DatetimeIndex) or a CSV
31        reload.  Must contain at least ``ticker``, ``Date`` (or index), and
32        ``Close``.
33
34    Returns
35    -------
36    pd.DataFrame
37        Columns limited to ``ticker, Date, Open, High, Low, Close, Volume``
38        with correct dtypes, tz-naive dates, and sorted by
39        ``[ticker, Date]``.
40
41    Raises
42    ------
43    ValueError
44        If the input is empty or essential columns are missing.
45    """
46    if df.empty:
47        raise ValueError("clean_prices: input DataFrame is empty.")
48
49    if "Date" not in df.columns:
50        df = df.reset_index()
51        df = df.rename(columns={"index": "Date"})
52
53    cols = [c for c in _PRICE_COLS if c in df.columns]
54    missing = {"ticker", "Date", "Close"} - set(cols)
55    if missing:
56        raise ValueError(f"clean_prices: missing required columns {missing}.")
57    df = df[cols].copy()
58
59    df["Date"] = pd.to_datetime(df["Date"]).dt.tz_localize(None)
60    for col in ("Open", "High", "Low", "Close", "Volume"):
61        if col in df.columns:
62            df[col] = pd.to_numeric(df[col], errors="coerce")
63
64    df = df.sort_values(["ticker", "Date"]).reset_index(drop=True)
65    log.info("clean_prices: %d rows, %d tickers.", len(df), df["ticker"].nunique())
66    return df

 79def clean_options(df: pd.DataFrame) -> pd.DataFrame:
 80    """Standardize a raw options-chain DataFrame.
 81
 82    Parameters
 83    ----------
 84    df : pd.DataFrame
 85        Raw output from :func:`get_options_chain`.  The column
 86        ``option_type`` is renamed to ``contractType`` if present.
 87
 88    Returns
 89    -------
 90    pd.DataFrame
 91        Columns limited to the canonical set, correct dtypes, rows with
 92        missing ``strike`` or ``impliedVolatility`` dropped, sorted by
 93        ``[ticker, expiration, contractType, strike]``.
 94
 95    Raises
 96    ------
 97    ValueError
 98        If the input is empty or essential columns are missing.
 99    """
100    if df.empty:
101        raise ValueError("clean_options: input DataFrame is empty.")
102
103    df = df.copy()
104    df = df.rename(columns={"option_type": "contractType"})
105
106    cols = [c for c in _OPTION_COLS if c in df.columns]
107    required = {"ticker", "expiration", "contractType", "strike", "impliedVolatility"}
108    missing = required - set(cols)
109    if missing:
110        raise ValueError(f"clean_options: missing required columns {missing}.")
111    df = df[cols]
112
113    df["expiration"] = pd.to_datetime(df["expiration"])
114    for col in ("strike", "bid", "ask", "volume", "openInterest", "impliedVolatility"):
115        if col in df.columns:
116            df[col] = pd.to_numeric(df[col], errors="coerce")
117
118    before = len(df)
119    df = df.dropna(subset=["strike", "impliedVolatility"])
120    dropped = before - len(df)
121    if dropped:
122        log.info("clean_options: dropped %d rows with missing strike/IV.", dropped)
123
124    df = df.sort_values(
125        ["ticker", "expiration", "contractType", "strike"]
126    ).reset_index(drop=True)
127    log.info("clean_options: %d rows retained.", len(df))
128    return df

23) -> tuple[pd.DataFrame, pd.DataFrame]:
24    """Compute put/call ratios by ticker and by ticker-expiration.
25
26    Parameters
27    ----------
28    options : pd.DataFrame
29        Cleaned options with ``ticker``, ``expiration``, ``contractType``,

32    Returns

39        ``days_to_exp``, ``horizon`` (near / mid / far).
40    """
41    # --- Aggregate by ticker ---
42    agg = (
43        options.groupby(["ticker", "contractType"])[["volume", "openInterest"]]
44        .sum()
45        .unstack("contractType")
46    )
47    agg.columns = ["_".join(c) for c in agg.columns]
48

50        "pcr_volume":  agg["volume_put"] / agg["volume_call"],
51        "pcr_oi":      agg["openInterest_put"] / agg["openInterest_call"],
52        "put_volume":  agg["volume_put"],

54        "put_oi":      agg["openInterest_put"],
55        "call_oi":     agg["openInterest_call"],
56    }).round(4)
57

59    agg_exp = (
60        options.groupby(["ticker", "expiration", "contractType"])[
61            ["volume", "openInterest"]
62        ]
63        .sum()
64        .unstack("contractType")
65        .fillna(0)
66    )
67    agg_exp.columns = ["_".join(c) for c in agg_exp.columns]
68
69    pcr_expiry = pd.DataFrame({
70        "pcr_volume": (
71            agg_exp["volume_put"]
72            / agg_exp["volume_call"].replace(0, np.nan)
73        ),
74        "pcr_oi": (
75            agg_exp["openInterest_put"]
76            / agg_exp["openInterest_call"].replace(0, np.nan)
77        ),
78    }).reset_index()
79
80    today = pd.Timestamp.today().normalize()
81    pcr_expiry["days_to_exp"] = (pcr_expiry["expiration"] - today).dt.days

83        pcr_expiry["days_to_exp"],

86    )
87

89        "compute_put_call_ratios: %d tickers, %d expiry rows.",
90        len(pcr_ticker), len(pcr_expiry),
91    )

def compute_iv_skew( options: pandas.DataFrame, dte_range: tuple[int, int] = (15, 60), atm_range: tuple[float, float] = (0.97, 1.03), otm_range: tuple[float, float] = (0.85, 0.97)) -> pandas.DataFrame: 
137def compute_iv_skew(
138    options: pd.DataFrame,
139    dte_range: tuple[int, int] = (15, 60),
140    atm_range: tuple[float, float] = (0.97, 1.03),
141    otm_range: tuple[float, float] = (0.85, 0.97),
142) -> pd.DataFrame:
143    """Compute implied-volatility skew per ETF.
144
145    IV Skew = median(OTM put IV) - median(ATM put IV), restricted to puts
146    within the specified DTE window.
147
148    Parameters
149    ----------
150    options : pd.DataFrame
151        Must contain ``ticker``, ``contractType``, ``impliedVolatility``,
152        ``moneyness``, and ``dte``.
153    dte_range : tuple[int, int]
154        (min, max) days to expiration, inclusive.
155    atm_range : tuple[float, float]
156        (lo, hi) moneyness bounds for ATM puts, inclusive.
157    otm_range : tuple[float, float]
158        (lo, hi) moneyness bounds for OTM puts. Lower inclusive, upper
159        exclusive.
160
161    Returns
162    -------
163    pd.DataFrame
164        Index = ticker.  Columns: ``atm_iv``, ``otm_iv``, ``iv_skew``,
165        ``n_atm``, ``n_otm``.
166    """
167    dte_lo, dte_hi = dte_range
168    atm_lo, atm_hi = atm_range
169    otm_lo, otm_hi = otm_range
170
171    puts = options[
172        (options["contractType"] == "put")
173        & (options["dte"].between(dte_lo, dte_hi))
174    ].copy()
175
176    puts["bucket"] = np.where(
177        puts["moneyness"].between(atm_lo, atm_hi),
178        "ATM",
179        np.where(
180            puts["moneyness"].between(otm_lo, otm_hi, inclusive="left"),
181            "OTM",
182            "other",
183        ),
184    )
185    puts = puts[puts["bucket"].isin(["ATM", "OTM"])]
186
187    rows = []
188    for ticker in sorted(options["ticker"].unique()):
189        t_puts = puts[puts["ticker"] == ticker]
190        atm = t_puts.loc[t_puts["bucket"] == "ATM", "impliedVolatility"]
191        otm = t_puts.loc[t_puts["bucket"] == "OTM", "impliedVolatility"]
192
193        atm_iv = atm.median() if len(atm) > 0 else np.nan
194        otm_iv = otm.median() if len(otm) > 0 else np.nan
195        iv_skew = (
196            otm_iv - atm_iv
197            if pd.notna(atm_iv) and pd.notna(otm_iv)
198            else np.nan
199        )
200
201        rows.append({
202            "ticker": ticker,
203            "atm_iv": atm_iv,
204            "otm_iv": otm_iv,
205            "iv_skew": iv_skew,
206            "n_atm": len(atm),
207            "n_otm": len(otm),
208        })
209
210    result = pd.DataFrame(rows).set_index("ticker")
211    log.info("compute_iv_skew: %d tickers.", len(result))
212    return result

280        w_norm = w / w.sum()
281
282        rows.append({
283            "ticker": etf,
284            "weighted_dte_earnings": (w_norm * d).sum(),
